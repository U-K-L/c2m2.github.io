248  private static Matrix ReactF(Matrix V, Matrix NN, Matrix MM, Matrix HH)
249  {
250  // The inputs are all column vectors, but I store them as column matrix
251  Matrix output = Matrix.Build.Dense(V.RowCount, V.ColumnCount);
252  Matrix prod = Matrix.Build.Dense(V.RowCount, V.ColumnCount);
253 
254  // Add current due to potassium
255  prod = NN.PointwisePower(4);
256  prod = (V - CellSolver2.ek).PointwiseMultiply(prod);
257  output = CellSolver2.gk * prod;
258 
259  // Add current due to sodium
260  prod = MM.PointwisePower(3);
261  prod = HH.PointwiseMultiply(prod);
262  prod = (V - CellSolver2.ena).PointwiseMultiply(prod);
263  output = output + CellSolver2.gna * prod;
264 
265  // Add leak current
266  output = output + CellSolver2.gl * (V - CellSolver2.el);
267 
268  // Return the negative of the total
269  return -1 * output;
270  }
