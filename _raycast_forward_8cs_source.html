110  private void TryRaycastHit()
111  {
112  // If the user enables "raycast mode" by VR controller or by mouse
113  bool raycastActive = OVRInput.Get(raycastButton, raycastController) || (mouseMode && Input.GetMouseButton(0));
114  if (raycastActive)
115  {
116  // Resolve raycast hit info for VR or mouse controller

119  if (mouseMode)
120  {

122  rayHit = Physics.Raycast(ray, out hit, 10f, layerMask);
123  }
124  else
125  {
126  RequestHandRenderState(true); // Enable raycasting hand
127  Vector3 globalForward = transform.TransformDirection(relativeForward); // Find the direction of the raycast
128  rayHit = Physics.Raycast(transform.position, globalForward, out hit, 10f, layerMask);
129  }
130  if (rayHit)
131  { // If our raycast hits a raycastable target,
132  // Are we raycasting to a new object, and does that object have any valid raycasting events?
133  valid = ResolveCurrentTarget(hit);
134  if (valid)
135  { // If we have a valid target to hit,
136  hitTriggered = false;
137  holdTriggered = false;
138  endTriggered = false;
139  // Don't allow calling more than one hit, hold, or end event in a single frame
140  DipointLinerendSetPositions(transform.position, hit.point); // Visualize our raycast as long as we have a valid target
141  ResolveClickEvents(hit); // Call events according to trigger button system
142  ResolveProximityEvents(hit); // Call events according to proximity distance activation system
143  // If we actually activated any triggers this frame, change hand asthetics to reflect it
144  if (holdTriggered) { RequestNewState(StateCode.HOLD); }
145  else if (hitTriggered) { RequestNewState(StateCode.HIT); }
146  else if (endTriggered) { RequestNewState(StateCode.END); }
147  else { RequestNewState(StateCode.NULL); } // Othwerwise we are inactive
148  }
149  else { DontDrawLinerend(); } // If we don't have a valid target, don't draw the linerender
150  }
151  else
152  {
153  DontDrawLinerend();
154  } // If our raycast did NOT hit a raycastable target, don't draw the linerender
155  }
156  else
157  { // If we aren't in "raycast mode"
158  RequestHandRenderState(false); // Return to default hand
159  // Turn off controller vibration & don't draw the raycast
160  RequestNewState(StateCode.NULL);
161  DontDrawLinerend();
162  if (touched) { touched = false; } // Allow proximity touches again
163  if (clicked) { clicked = false; } // Allow trigger button activation again
164  }

167  // Are we raycasting to a new object, and does that object have any valid raycasting events?
168  bool ResolveCurrentTarget(RaycastHit hit)
169  {
170  int newID = hit.collider.gameObject.GetInstanceID();
171  if (newID != activeObjectID)
172  { // If we have found a new object,
173  RaycastEventManager newTriggerManager = hit.collider.GetComponentInParent<RaycastEventManager>();
174  if (newTriggerManager != null)
175  { // If it has a valid trigger,
176  // Find whether or not there are valid hit, hold, and end events based on handedness
177  if (rightHand)
178  {
179  validHitEvent = (newTriggerManager.rightTrigger.OnPress.GetPersistentEventCount() > 0) ? true : false;
180  validHoldEvent = (newTriggerManager.rightTrigger.OnHoldPress.GetPersistentEventCount() > 0) ? true : false;
181  validEndEvent = (newTriggerManager.rightTrigger.OnEndPress.GetPersistentEventCount() > 0) ? true : false;
182  }
183  else
184  {
185  validHitEvent = (newTriggerManager.leftTrigger.OnPress.GetPersistentEventCount() > 0) ? true : false;
186  validHoldEvent = (newTriggerManager.leftTrigger.OnHoldPress.GetPersistentEventCount() > 0) ? true : false;
187  validEndEvent = (newTriggerManager.leftTrigger.OnEndPress.GetPersistentEventCount() > 0) ? true : false;
188  }
189  if (validHitEvent || validHoldEvent || validEndEvent)
190  { // If we have found some active event to activate, save it and mark this target as valid
191  activeTriggerManager = newTriggerManager;
192  activeObjectID = newID;
193  return true;
194  }
195  }
196  // If the object doesn't have a RaycastTriggerManager, then we don't have a valid object
197  return false;
198  }
199  // Otherwise we have a previously valid object
200  else return false;

250  // Functionality for invoking trigger events for close-proximity raycast hits
251  void ResolveProximityEvents(RaycastHit hit)
252  {
253  if (hit.distance < minHitDist)
254  { // If we are within close enough to activate our trigger,
255  if (!touched)
256  { // & if we didn't just activate that trigger by being within distance already,
257  touched = true; // Don't reactivate any triggers until we move far enough away.
258  AttemptHitEvent(hit);
259  }
260  else { AttemptHoldEvent(hit); } // If we're still close, but have already trigegred the hit event, then we're holding
261  }
262  else if (hit.distance < hysteresisDist)
263  { // If we are still within hysteresis range, don't allow further touches but continue calling hold events
264  if (touched) { AttemptHoldEvent(hit); } // If we've already hit the object, then we can hold it
265  }
266  else
267  { // If we are far enough away, then we are done with our proximity touch/hold
268  if (touched)
269  {
270  touched = false; // Allow proximity touches again
271  AttemptEndEvent(hit);
272  }
273  }
