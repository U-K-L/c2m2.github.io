268  private static Vector reactF(Vector V, Vector NN, Vector MM, Vector HH)
269  {
270  Vector output = Vector.Build.Dense(V.Count);
271  Vector prod = Vector.Build.Dense(V.Count);
272 
273  // Add current due to potassium
274  prod = NN.PointwisePower(4);
275  prod = (V - ek).PointwiseMultiply(prod);
276  output = gk * prod;
277 
278  // Add current due to sodium
279  prod = MM.PointwisePower(3);
280  prod = HH.PointwiseMultiply(prod);
281  prod = (V - ena).PointwiseMultiply(prod);
282  output = output + gna * prod;
283 
284  // Add leak current
285  output = output + gl * (V - el);
286 
287  // Return the negative of the total
288  return -1 * output;
289  }
