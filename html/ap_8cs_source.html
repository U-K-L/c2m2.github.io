674  {

677  if( rows(a)!=cols(a) )

680  if( n==0 )
681  return true;
682  mx = 0;

684  for( i=0; i<n; i++)
685  {
686  for(j=i+1; j<n; j++)
687  {

711  matrix is considered Hermitian

714  {
715  int i, j, n;

717  complex v1, v2, vt;
718  if( rows(a)!=cols(a) )
719  return false;

721  if( n==0 )
722  return true;

724  err = 0;

923  /*
924  * CSV operations: reading CSV file to real matrix.
925  *
926  * This function reads CSV file and stores its contents to double
927  * precision 2D array. Format of the data file must conform to RFC 4180
928  * specification, with additional notes:
929  * - file size should be less than 2GB
930  * - ASCI encoding, UTF-8 without BOM (in header names) are supported
931  * - any character (comma/tab/space) may be used as field separator, as
932  * long as it is distinct from one used for decimal point
933  * - multiple subsequent field separators (say, two spaces) are treated
934  * as MULTIPLE separators, not one big separator
935  * - both comma and full stop may be used as decimal point. Parser will
936  * automatically determine specific character being used. Both fixed
937  * and exponential number formats are allowed. Thousand separators
938  * are NOT allowed.
939  * - line may end with \n (Unix style) or \r\n (Windows style), parser
940  * will automatically adapt to chosen convention
941  * - escaped fields (ones in double quotes) are not supported
942  *
943  * INPUT PARAMETERS:
944  * filename relative/absolute path
945  * separator character used to separate fields. May be ' ',
946  * ',', '\t'. Other separators are possible too.
947  * flags several values combined with bitwise OR:
948  * * alglib::CSV_SKIP_HEADERS - if present, first row
949  * contains headers and will be skipped. Its
950  * contents is used to determine fields count, and
951  * that's all.
952  * If no flags are specified, default value 0x0 (or
953  * alglib::CSV_DEFAULT, which is same) should be used.
954  *
955  * OUTPUT PARAMETERS:
956  * out 2D matrix, CSV file parsed with atof()
957  *
958  * HANDLING OF SPECIAL CASES:
959  * - file does not exist - alglib::ap_error exception is thrown
960  * - empty file - empty array is returned (no exception)
961  * - skip_first_row=true, only one row in file - empty array is returned
962  * - field contents is not recognized by atof() - field value is replaced
963  * by 0.0
964  */
965  public static void read_csv(string filename, char separator, int flags, out double[,] matrix)
966  {
967  //
968  // Parameters
969  //
970  bool skip_first_row = (flags&CSV_SKIP_HEADERS)!=0;
971 
972  //
973  // Prepare empty output array
974  //
975  matrix = new double[0,0];
976 
977  //
978  // Read file, normalize file contents:
979  // * replace 0x0 by spaces
980  // * remove trailing spaces and newlines
981  // * append trailing '\n' and '\0' characters
982  // Return if file contains only spaces/newlines.
983  //
984  byte b_space = System.Convert.ToByte(' ');
985  byte b_tab = System.Convert.ToByte('\t');
986  byte b_lf = System.Convert.ToByte('\n');
987  byte b_cr = System.Convert.ToByte('\r');
988  byte b_comma = System.Convert.ToByte(',');

991  if( v0.Length==0 )
992  return;

994  int filesize = v0.Length;
995  for(int i=0; i<filesize; i++)
996  v1[i] = v0[i]==0 ? b_space : v0[i];
997  for(; filesize>0; )
998  {
999  byte c = v1[filesize-1];
1000  if( c==b_space || c==b_tab || c==b_cr || c==b_lf )
1001  {
1002  filesize--;
1003  continue;
1004  }
1005  break;
1006  }
1007  if( filesize==0 )
1008  return;
1009  v1[filesize+0] = b_lf;
1010  v1[filesize+1] = 0x0;
1011  filesize+=2;
1012 
1013 
1014  //
1015  // Scan dataset.
1016  //
1017  int rows_count, cols_count, max_length = 0;
1018  cols_count = 1;
1019  for(int idx=0; idx<filesize; idx++)
1020  {
1021  if( v1[idx]==separator )
1022  cols_count++;
1023  if( v1[idx]==b_lf )
1024  break;
1025  }
1026  rows_count = 0;
1027  for(int idx=0; idx<filesize; idx++)
1028  if( v1[idx]==b_lf )
1029  rows_count++;
1030  if( rows_count==1 && skip_first_row ) // empty output, return
1031  return;
1032  int[] offsets = new int[rows_count*cols_count];
1033  int[] lengths = new int[rows_count*cols_count];
1034  int cur_row_idx = 0;
1035  for(int row_start=0; v1[row_start]!=0x0; )
1036  {
1037  // determine row length
1038  int row_length;
1039  for(row_length=0; v1[row_start+row_length]!=b_lf; row_length++);
1040 
1041  // determine cols count, perform integrity check
1042  int cur_cols_cnt=1;
1043  for(int idx=0; idx<row_length; idx++)
1044  if( v1[row_start+idx]==separator )
1045  cur_cols_cnt++;
1046  if( cols_count!=cur_cols_cnt )
1047  throw new alglib.alglibexception("read_csv: non-rectangular contents, rows have different sizes");
1048 
1049  // store offsets and lengths of the fields
1050  int cur_offs = 0;
1051  int cur_col_idx = 0;
1052  for(int idx=0; idx<row_length+1; idx++)
1053  if( v1[row_start+idx]==separator || v1[row_start+idx]==b_lf )
1054  {
1055  offsets[cur_row_idx*cols_count+cur_col_idx] = row_start+cur_offs;
1056  lengths[cur_row_idx*cols_count+cur_col_idx] = idx-cur_offs;
1057  max_length = idx-cur_offs>max_length ? idx-cur_offs : max_length;
1058  cur_offs = idx+1;
1059  cur_col_idx++;
1060  }
1061 
1062  // advance row start
1063  cur_row_idx++;
1064  row_start = row_start+row_length+1;
1065  }
1066 
1067  //
1068  // Convert
1069  //
1070  int row0 = skip_first_row ? 1 : 0;
1071  int row1 = rows_count;

1073  matrix = new double[row1-row0, cols_count];
1074  alglib.AE_CRITICAL_ASSERT(culture.NumberFormat.NumberDecimalSeparator==".");
1075  for(int ridx=row0; ridx<row1; ridx++)
1076  for(int cidx=0; cidx<cols_count; cidx++)
1077  {
1078  int field_len = lengths[ridx*cols_count+cidx];
1079  int field_offs = offsets[ridx*cols_count+cidx];
1080 
1081  // replace , by full stop
1082  for(int idx=0; idx<field_len; idx++)
1083  if( v1[field_offs+idx]==b_comma )
1084  v1[field_offs+idx] = b_fullstop;
1085 
1086  // convert
1087  string s_val = System.Text.Encoding.ASCII.GetString(v1, field_offs, field_len);
1088  double d_val;
1089  Double.TryParse(s_val, System.Globalization.NumberStyles.Float, culture, out d_val);
1090  matrix[ridx-row0,cidx] = d_val;
1091  }
