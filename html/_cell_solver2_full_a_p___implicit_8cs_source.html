312  private static Vector reactF(Vector V, Vector NN, Vector MM, Vector HH, List<int> bcIndices)
313  {
314  Vector output = Vector.Build.Dense(V.Count);
315  Vector prod = Vector.Build.Dense(V.Count);
316 
317  // Add current due to potassium
318  prod = NN.PointwisePower(4);
319  prod = (V - ek).PointwiseMultiply(prod);
320  output = gk * prod;
321 
322  // Add current due to sodium
323  prod = MM.PointwisePower(3);
324  prod = HH.PointwiseMultiply(prod);
325  prod = (V - ena).PointwiseMultiply(prod);
326  output = output + gna * prod;
327 
328  // Add leak current
329  output = output + gl * (V - el);
330 
331  // Return the negative of the total
332  output.Multiply(-1, output);
333 
334  output.SetSubVector(0, V.Count, boundaryConditions(output, bcIndices));
335 
336  return output;
