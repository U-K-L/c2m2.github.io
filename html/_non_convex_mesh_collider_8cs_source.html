56  // Find RigidBody & store its settings

58  if (rb == null) throw new RigidbodyNotFoundException();

61  bool rbUsedGravity = rb.useGravity;
62  rb.useGravity = false;
63  bool rbWasKinematic = rb.isKinematic;
64  rb.isKinematic = true;
65 
66  if (!createChild)
67  {
68  BoxCollider[] oldCols = go.GetComponents<BoxCollider>();
69  for (int i = 0; i < oldCols.Length; i++)

76  GameObject tempParent = new GameObject("Temp_CompoundColliderParent");
77  go.transform.parent = tempParent.transform;
78  // Normalize position, rotation, and scale and store the old values
79  Vector3 localPos = go.transform.localPosition;
80  go.transform.localPosition = Vector3.zero;

82  go.transform.localRotation = Quaternion.Euler(Vector3.zero);

93  //merge boxes to create fewer, larger box colliders
94  Box[] mergedBoxes = mergeBoxes ? MergeBoxes(boxes.ToArray()) : boxes;
95 
96  boxColliders = new BoxCollider[mergedBoxes.Length];
97  for (int i = 0; i < mergedBoxes.Length; i++)
98  {
99  BoxCollider bc = (createChild ? collidersGo : go).AddComponent<BoxCollider>();
100  bc.size = mergedBoxes[i].Size;
101  bc.center = mergedBoxes[i].Center;
102  boxColliders[i] = bc;
103  }
104  Debug.Log("NonConvexMeshCollider: " + mergedBoxes.Length + " box colliders created");
105  collidersGo.name = "Colliders " + boxesPerEdge + " (" + mergedBoxes.Length + " colliders)";
106 
107  //cleanup stuff not needed anymore on collider child obj
108  UnityEngine.Object.Destroy(collidersGo.GetComponent<MeshFilter>());
109  UnityEngine.Object.Destroy(collidersGo.GetComponent<MeshCollider>());
110  UnityEngine.Object.Destroy(collidersGo.GetComponent<Rigidbody>());
111  if (!createChild)
112  UnityEngine.Object.Destroy(collidersGo);
113  }
114  finally
115  {
116  //reset original state of root object
117  go.transform.parent = pTransform;
118  go.transform.localPosition = localPos;
119  go.transform.localRotation = localRot;
120  go.transform.localScale = localScale;
121  if (rbWasKinematic)

123  if (rbUsedGravity)
124  rb.useGravity = true;
125  UnityEngine.Object.Destroy(tempParent);
126  }

132  private static Box[] MergeBoxes(Box[] boxes)
133  {
134  var mergeDirections = new[] {
135  new Vector3Int(1,0,0),
136  new Vector3Int(0,1,0),
137  new Vector3Int(0,0,1),
138  new Vector3Int(-1,0,0),
139  new Vector3Int(0,-1,0),
140  new Vector3Int(0,0,-1),
141  };
142  var foundSomethingToMerge = false;
143  do
144  {
145  foreach (var mergeDirection in mergeDirections)
146  {
147  foundSomethingToMerge = false;
148  foreach (var box in boxes)
149  {
150  var merged = box.TryMerge(mergeDirection);
151  if (merged)
152  foundSomethingToMerge = true;
153  }
154  boxes = boxes.Select(b => b.Root).Distinct().ToArray();
155  }
156  } while (foundSomethingToMerge);
157  return boxes.Select(b => b.Root).Distinct().ToArray();
158  }

197  private static IEnumerable<Box> CreateMeshIntersectingBoxes(GameObject colliderGo, int boxesPerEdge)
198  {
199  var go = colliderGo.transform.parent.gameObject;
200  var bounds = CalculateLocalBounds(go);
201 
202  var boxes = new Box[boxesPerEdge, boxesPerEdge, boxesPerEdge];
203  var boxColliderPositions = new bool[boxesPerEdge, boxesPerEdge, boxesPerEdge];
204  var s = bounds.size / boxesPerEdge;
205  var halfExtent = s / 2;
206 
207  for (var x = 0; x < boxesPerEdge; x++)
208  {
209  for (var y = 0; y < boxesPerEdge; y++)
210  {
211  for (var z = 0; z < boxesPerEdge; z++)
212  {
213  var center = new Vector3(
214  bounds.center.x - bounds.size.x / 2 + s.x * x + s.x / 2,
215  bounds.center.y - bounds.size.y / 2 + s.y * y + s.y / 2,
216  bounds.center.z - bounds.size.z / 2 + s.z * z + s.z / 2);
217  var colliders = Physics.OverlapBox(center, halfExtent);
218  if (colliders.Length > 0 && colliders.Any(c => c.gameObject == colliderGo))
219  boxColliderPositions[x, y, z] = true;
220  }
221  }
222  }
223 
224  for (var x = 0; x < boxesPerEdge; x++)
225  {
226  for (var y = 0; y < boxesPerEdge; y++)
227  {
228  for (var z = 0; z < boxesPerEdge; z++)
229  {
230  if (!boxColliderPositions[x, y, z]) continue;
231  var center = new Vector3(
232  bounds.center.x - bounds.size.x / 2 + s.x * x + s.x / 2,
233  bounds.center.y - bounds.size.y / 2 + s.y * y + s.y / 2,
234  bounds.center.z - bounds.size.z / 2 + s.z * z + s.z / 2);
235  var b = new Box(boxes, center, s, new Vector3Int(x, y, z));
236  boxes[x, y, z] = b;
237  yield return b;
238  }
239  }
240  }
241  }

340  public bool TryMerge(Vector3Int direction)
341  {
342  if (Parent != null) return false;
343  foreach (var p in CoveredGridPositions)
344  {
345  var pos = new Vector3Int(p.X + direction.X, p.Y + direction.Y, p.Z + direction.Z);
346  if (pos.X < 0 || pos.Y < 0 || pos.Z < 0)
347  continue;
348  if (pos.X >= boxes.GetLength(0) || pos.Y >= boxes.GetLength(1) || pos.Z >= boxes.GetLength(2))
349  continue;
350  var b = boxes[pos.X, pos.Y, pos.Z];
351  if (b == null)
352  continue;
353  b = b.Root;
354  if (b == this)
355  continue;
356  if (direction.X == 0 && b.GridSize.X != GridSize.X)
357  continue;
358  if (direction.Y == 0 && b.GridSize.Y != GridSize.Y)
359  continue;
360  if (direction.Z == 0 && b.GridSize.Z != GridSize.Z)
361  continue;
362  if (direction.X == 0 && MinGridPos.X != b.MinGridPos.X)
363  continue;
364  if (direction.Y == 0 && MinGridPos.Y != b.MinGridPos.Y)
365  continue;
366  if (direction.Z == 0 && MinGridPos.Z != b.MinGridPos.Z)
367  continue;
368  MergeWith(b);
369  return true;
370  }
371  return false;
372  }
