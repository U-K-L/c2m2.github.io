193  private void BuildAdjacencyList()
195  {
196  int space = 6 * (subdivisions + 1); // Reserve 6 places if no subdivisions, 12 places if 1, etc
197  for (int i = 0; i < uniqueVerts.Length; i++)
198  { // Create a new list for each adjacency list index to store adjacent info

200  }
201  Vector3 vec0, vec1, vec2;
202  int v0, v1, v2, v01current, v01previous, v12current, v12previous, v20current, v20previous;
203  float scaler, dist01, dist12, dist20, dist01t2, dist12t0, dist20t1;
204  if (subdivisions == 0)
205  { // If we DON'T want to add any invisible edges, just add the triangle edges
206  for (int i = 0; i < triangles.Length; i += 3)
207  { // For each triangle,
208  // Find the Vector3 position of each point in the triangle,
209  vec0 = vertices[triangles[i]];
210  vec1 = vertices[triangles[i + 1]];
211  vec2 = vertices[triangles[i + 2]];
212  // Find the UNIQUE vertex that occupies that same Vector3,
213  v0 = uniqueVertReverseLookup[vec0];
214  v1 = uniqueVertReverseLookup[vec1];
215  v2 = uniqueVertReverseLookup[vec2];
216  // Calculate the distance between each unqiue point and store them in the adjacency matrix
217  dist01 = Vector3.Distance(vec0, vec1);
218  dist12 = Vector3.Distance(vec1, vec2);
219  dist20 = Vector3.Distance(vec2, vec0);
220  AdjacencyListAddEdge(v0, v1, dist01);
221  AdjacencyListAddEdge(v1, v2, dist12);
222  AdjacencyListAddEdge(v2, v0, dist20);
223  }
224  }
225  else
226  { // If we DO want to add invisible edges,
227  for (int i = 0; i < triangles.Length; i += 3)
228  { // For each triangle, find the points of the triange, find the UNIQUE vertex that occupies that same spot, and add the unique vertices as new edges
229  // Get the vector of each vertex in the triangle
230  vec0 = vertices[triangles[i]];
231  vec1 = vertices[triangles[i + 1]];
232  vec2 = vertices[triangles[i + 2]];
233  // Get the UNIQUE index of each vertex in the triangle
234  v0 = uniqueVertReverseLookup[vec0];
235  v1 = uniqueVertReverseLookup[vec1];
236  v2 = uniqueVertReverseLookup[vec2];
237  // Add the default edges to the graph so that we can skip invisible edges later if we want to
238  dist01 = Vector3.Distance(vec0, vec1);
239  dist12 = Vector3.Distance(vec1, vec2);
240  dist20 = Vector3.Distance(vec2, vec0);
241  AdjacencyListAddEdge(v0, v1, dist01);
242  AdjacencyListAddEdge(v1, v2, dist12);
243  AdjacencyListAddEdge(v2, v0, dist20);
244  // We're going to "crawl" along each edge 01, 12, and 20 of this triangle and add edges connecting each intermediate vertex with the previous one AND the opposite one
245  v01previous = v0;
246  v12previous = v1;
247  v20previous = v2;
248  // Get the distance between each invisible vertex on the same edge, since they will be placed along even fractions of the edge
249  dist01 *= divider;
250  dist12 *= divider;
251  dist20 *= divider;
252  // Go through each subdivision
253  scaler = divider;
254  for (int s = 0; s < subdivisions; s++)
255  { // Go along each subdivided region from (1/4 to 2/4 to 3/4 to finally 4/4
256  // Recalculate the inbetween positions
257  // Get the index of each in between vertex from its position
258  v01current = uniqueVertReverseLookup[Vector3.Lerp(vec0, vec1, scaler)];
259  v12current = uniqueVertReverseLookup[Vector3.Lerp(vec1, vec2, scaler)];
260  v20current = uniqueVertReverseLookup[Vector3.Lerp(vec2, vec0, scaler)];
261  // Calculate distance from each in between vertex to its opposite
262  dist01t2 = Vector3.Distance(uniqueVerts[v01current], vec2);
263  dist12t0 = Vector3.Distance(uniqueVerts[v12current], vec0);
264  dist20t1 = Vector3.Distance(uniqueVerts[v20current], vec1);
265  // Add an edge between "current" and its opposite vert AND add THE BACKWARDS EDGE
266  AdjacencyListAddEdge(v01current, v2, dist01t2);
267  AdjacencyListAddEdge(v12current, v0, dist12t0);
268  AdjacencyListAddEdge(v20current, v1, dist20t1);
269  // Add an edge between "previous" and "current" AND add THE BACKWARDS EDGE
270  AdjacencyListAddEdge(v01previous, v01current, dist01);
271  AdjacencyListAddEdge(v12previous, v12current, dist12);
272  AdjacencyListAddEdge(v20previous, v20current, dist20);
273  // Slide down the edge
274  v01previous = v01current;
275  v12previous = v12current;
276  v20previous = v20current;
277  // Slide up the Lerp scaler
278  scaler += divider;
279  }
280  // Now we're at the end, so just connect the second to last invisible vert to the opposite vert. Don't connect to the opposite vert because there will be more invisible verts along that edge
281  AdjacencyListAddEdge(v01previous, v1, dist01);
282  AdjacencyListAddEdge(v2, v12previous, dist12);
283  AdjacencyListAddEdge(v0, v20previous, dist20);
284  }

293  {

296  { // If we have never added point [end] to point [origin]'s list, then the back edge should not exist either, so we only need to check one.
