270  private static Vector reactF(Vector V, Vector NN, Vector MM, Vector HH)
271  {
272  Vector output = Vector.Build.Dense(V.Count);
273  Vector prod = Vector.Build.Dense(V.Count);
274 
275  // Add current due to potassium
276  prod = NN.PointwisePower(4);
277  prod = (V - ek).PointwiseMultiply(prod);
278  output = gk * prod;
279 
280  // Add current due to sodium
281  prod = MM.PointwisePower(3);
282  prod = HH.PointwiseMultiply(prod);
283  prod = (V - ena).PointwiseMultiply(prod);
284  output = output + gna * prod;
285 
286  // Add leak current
287  output = output + gl * (V - el);
288 
289  // Return the negative of the total
290  return -1 * output;
291  }
